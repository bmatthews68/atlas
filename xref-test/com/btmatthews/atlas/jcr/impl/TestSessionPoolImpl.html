3   import org.apache.commons.pool.KeyedPoolableObjectFactory;
4   import org.junit.Before;
5   import org.junit.Test;
6   import org.mockito.Mock;
7   
8   import javax.jcr.Session;

10  import static org.junit.Assert.assertSame;
11  import static org.mockito.Matchers.eq;
12  import static org.mockito.Mockito.*;
13  import static org.mockito.MockitoAnnotations.initMocks;
14  
15  public class TestSessionPoolImpl {
16  
17      private static final String DEFAULT_WORKSPACE_NAME = "default";
18      @Mock
19      private KeyedPoolableObjectFactory<String, Session> poolableSessionFactory;
20      @Mock
21      private Session pooledSession;
22  
23      @Before
24      public void init() {
25          initMocks(this);
26      }
27  
28      @Test
29      public void borrowAndReturnASingleSession() throws Exception {
30          final PooledSessionFactory sessionPool = new PooledSessionFactory(poolableSessionFactory);
31          when(poolableSessionFactory.makeObject(DEFAULT_WORKSPACE_NAME)).thenReturn(pooledSession);
32          final Session session = sessionPool.getSession(DEFAULT_WORKSPACE_NAME);
33          assertSame(pooledSession, session);
34          sessionPool.releaseSession(DEFAULT_WORKSPACE_NAME, session);
35          sessionPool.shutdown();
36          verify(poolableSessionFactory).makeObject(eq(DEFAULT_WORKSPACE_NAME));
37          verify(poolableSessionFactory).activateObject(eq(DEFAULT_WORKSPACE_NAME), same(session));
38          verify(poolableSessionFactory).passivateObject(eq(DEFAULT_WORKSPACE_NAME), same(pooledSession));
39          verify(poolableSessionFactory).destroyObject(eq(DEFAULT_WORKSPACE_NAME), same(pooledSession));
40          verifyNoMoreInteractions(poolableSessionFactory);
41          verifyZeroInteractions(pooledSession);
42      }
43  }
