30  public class TestTenantFilter {
31  
32      @Mock
33      private Tenant tenant;
34      @Mock
35      private TenantService service;
36      @Mock
37      private HttpServletRequest request;
38      @Mock
39      private HttpServletResponse response;
40      @Mock
41      private FilterConfig filterConfig;
42      @Mock
43      private FilterChain chain;
44      private Filter filter;
45  
46      @Before
47      public void setup() throws Exception {
48          initMocks(this);
49          filter = new TenantFilter(service, "X-Atlas-Tenant");
50          filter.init(filterConfig);
51      }
52  
53      @After
54      public void teardown() {
55          filter.destroy();
56      }
57  
58      @Test
59      public void shouldSetTenantContextWhenTenantHeaderSet() throws Exception {
60          when(request.getHeader(eq("X-Atlas-Tenant"))).thenReturn("shop");
61          when(service.lookupTenantByKey(anyString())).thenReturn(tenant);
62          filter.doFilter(request, response, chain);
63          assertThat(TenantContextHolder.peek(), is(nullValue()));
64          verify(request).getHeader(eq("X-Atlas-Tenant"));
65          verify(service).lookupTenantByKey(eq("shop"));
66          verify(chain).doFilter(same(request), same(response));
67          verifyNoMoreInteractions(service, request, response, tenant, chain);
68      }
69  }
