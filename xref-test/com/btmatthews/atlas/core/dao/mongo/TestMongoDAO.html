34  /**
35   * Unit test the {@link MongoDAO} data access object implementation.
36   *
37   * @author <a href="mailto:brian@btmatthews.com">Brian Thomas Matthews</a>
38   * @since 1.0.0
39   */
40  public class TestMongoDAO {
41  
42      /**
43       * Used to collect test failures within a single test case.
44       */
45      @Rule
46      public ErrorCollector collector = new ErrorCollector();
47      /**
48       * The data access object being tested.
49       */
50      private DAO<Person> dao;
51      /**
52       * Used to mock the Mongo data store.
53       */
54      private Fongo fongo = new Fongo("localhost");
55  
56      /**
57       * Prepare for test case execution.
58       */
59      @Before
60      public void setup() {
61          final MongoDAO<Person, PersonImpl> mongoDao = new MongoDAO<Person, PersonImpl>(PersonImpl.class) {
62          };
63          mongoDao.setMongoDbFactory(new MongoDbFactory() {
64              @Override
65              public DB getDb() throws DataAccessException {
66                  return fongo.getDB("admin");
67              }
68  
69              @Override
70              public DB getDb(final String name) throws DataAccessException {
71                  return fongo.getDB(name);
72              }
73  
74              @Override
75              public PersistenceExceptionTranslator getExceptionTranslator() {
76                  return null;
77              }
78          });
79          dao = mongoDao;
80      }
81  
82      /**
83       * Make sure the {@link MongoDAO#find(com.btmatthews.atlas.core.common.Paging)} throws an {@link IllegalArgumentException} if {@code null} is
84       * passed as the {@code paging} parameter.
85       *
86       * @throws Exception An {@link IllegalArgumentException} is expected.
87       */
88      @Test(expected = IllegalArgumentException.class)
89      public void findWithNullPagingShouldFail() throws Exception {
90          dao.find(null);
91      }
92  
93      /**
94       * Test the {@link com.btmatthews.atlas.core.dao.mongo.MongoDAO#count()},
95       * {@link com.btmatthews.atlas.core.dao.mongo.MongoDAO#find(com.btmatthews.atlas.core.common.Paging)},
96       * {@link com.btmatthews.atlas.core.dao.mongo.MongoDAO#create(Object)},
97       * {@link com.btmatthews.atlas.core.dao.mongo.MongoDAO#read(String)},
98       * {@link com.btmatthews.atlas.core.dao.mongo.MongoDAO#update(Object)} and
99       * {@link com.btmatthews.atlas.core.dao.mongo.MongoDAO#destroy(Object)} methods.
100      */
101     @Test
102     public void checkFullObjectLifecycle() {
103         final Paging paging = new PagingBuilder().setPageNumber(0).setPageSize(100).build();
104         collector.checkThat(dao.count(), is(equalTo(0L)));
105         collector.checkThat(dao.find(paging).size(), is(equalTo(0)));
106         final Person person1 = new PersonImpl("ee749160-c6a0-11e2-8b8b-0800200c9a66", "Brian Matthews");
107         dao.create(person1);
108         collector.checkThat(dao.count(), is(equalTo(1L)));
109         collector.checkThat(dao.find(paging).size(), is(equalTo(1)));
110         final Person person2 = dao.read("ee749160-c6a0-11e2-8b8b-0800200c9a66");
111         collector.checkThat(person2, is(not(nullValue())));
112         collector.checkThat(person2, hasProperty("id", is(equalTo("ee749160-c6a0-11e2-8b8b-0800200c9a66"))));
113         collector.checkThat(person2, hasProperty("name", is(equalTo("Brian Matthews"))));
114         collector.checkThat(dao.count(), is(equalTo(1L)));
115         final Person person3 = new PersonImpl("ee749160-c6a0-11e2-8b8b-0800200c9a66", "Brian Thomas Matthews");
116         dao.update(person3);
117         final Person person4 = dao.read("ee749160-c6a0-11e2-8b8b-0800200c9a66");
118         collector.checkThat(person4, is(not(nullValue())));
119         collector.checkThat(person4, hasProperty("id", is(equalTo("ee749160-c6a0-11e2-8b8b-0800200c9a66"))));
120         collector.checkThat(person4, hasProperty("name", is(equalTo("Brian Thomas Matthews"))));
121         dao.destroy(person4);
122         collector.checkThat(dao.count(), is(equalTo(0L)));
123         collector.checkThat(dao.find(paging).size(), is(equalTo(0)));
124     }

159     /**
160      * Make sure the {@link MongoDAO#read(String)} method returns {@code null} if the object does not already exist
161      * in the data store.
162      */
163     @Test

167     }
