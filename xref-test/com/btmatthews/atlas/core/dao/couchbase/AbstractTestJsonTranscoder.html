34          final Person person = new PersonImpl(
35                  PersonTestData.PERSON_ID,
36                  PersonTestData.PERSON_NAME,
37                  PersonTestData.PERSON_EMAIL);
38          final CachedData cachedData = transcoder.encode(person);
39          assertNotNull(cachedData);
40          assertNotNull(cachedData.getData());
41          with(new ByteArrayInputStream(cachedData.getData()))
42                  .assertThat("$.id", equalTo(PersonTestData.PERSON_ID))
43                  .assertThat("$.name", equalTo(PersonTestData.PERSON_NAME))
44                  .assertThat("$.email", equalTo(PersonTestData.PERSON_EMAIL));
