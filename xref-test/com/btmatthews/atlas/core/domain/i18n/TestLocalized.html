17  package com.btmatthews.atlas.core.domain.i18n;
18  
19  import org.junit.Before;
20  import org.junit.Rule;
21  import org.junit.Test;
22  import org.junit.rules.ErrorCollector;
23  
24  import java.util.HashMap;
25  import java.util.Locale;
26  import java.util.Map;
27  
28  import static org.hamcrest.CoreMatchers.*;
29  
30  /**
31   * Unit test the {@link Localized} object.
32   *
33   * @author <a href="mailto:brian@btmatthews.com">Brian Thomas Matthews</a>
34   * @since 1.0.0
35   */
36  public class TestLocalized {
37  
38      private static final String HELLO_WORLD_ENGLISH = "Hello World!";
39      private static final String HELLO_WORLD_GERMAN = "Hallo Wert!";
40      private static final String GOODBYE_WORLD_ENGLISH = "Goodbye World!";
41      private static final String GOODBYE_WORLD_GERMAN = "Auf Wiedersehen, Welt!";
42      @Rule
43      public ErrorCollector collector = new ErrorCollector();
44      private Localized<String> single;
45      private Localized<String> single_copy;
46      private Localized<String> single_different;
47      private Localized<String> multi;
48      private Localized<String> multi_copy;
49      private Localized<String> multi_different;
50  
51      @Before
52      public void setup() {
53          single = new Localized<String>(Locale.ENGLISH, HELLO_WORLD_ENGLISH);
54          single_copy = new Localized<String>(Locale.ENGLISH, HELLO_WORLD_ENGLISH);
55          single_different = new Localized<String>(Locale.ENGLISH, GOODBYE_WORLD_ENGLISH);
56          Map<Locale, String> values = new HashMap<Locale, String>();
57          values.put(Locale.ENGLISH, HELLO_WORLD_ENGLISH);
58          values.put(Locale.GERMAN, HELLO_WORLD_GERMAN);
59          multi = new Localized<String>(values);
60          multi_copy = new Localized<String>(values);
61          values = new HashMap<Locale, String>();
62          values.put(Locale.ENGLISH, GOODBYE_WORLD_ENGLISH);
63          values.put(Locale.GERMAN, GOODBYE_WORLD_GERMAN);
64          multi_different = new Localized<String>(values);
65      }
66  
67      @Test
68      public void createSingleLocalizedValue() {
69          collector.checkThat(single.getValue(Locale.ENGLISH), is(equalTo(HELLO_WORLD_ENGLISH)));
70          collector.checkThat(single.getValues().size(), is(equalTo(1)));
71          collector.checkThat(single.getValues().get(Locale.ENGLISH), is(equalTo(HELLO_WORLD_ENGLISH)));
72          collector.checkThat(single.hashCode(), is(equalTo(-1006767329)));
73          collector.checkThat(single_copy.hashCode(), is(equalTo(-1006767329)));
74          collector.checkThat(single_different.hashCode(), is(equalTo(1598763198)));
75          collector.checkThat(single, is(equalTo(single)));
76          collector.checkThat(single, is(equalTo(single_copy)));
77          collector.checkThat(single, is(not(equalTo(null))));
78          collector.checkThat(single, is(not(equalTo(single_different))));
79          collector.checkThat(single.toString(), is(equalTo("Localized[{en=" + HELLO_WORLD_ENGLISH + "}]")));
80          collector.checkThat(single_copy.toString(), is(equalTo("Localized[{en=" + HELLO_WORLD_ENGLISH + "}]")));
81          collector.checkThat(single_different.toString(), is(equalTo("Localized[{en=" + GOODBYE_WORLD_ENGLISH + "}]")));
82      }
83  
84      @Test
85      public void createWithMultipleLocalizedValues() {
86          collector.checkThat(multi.getValue(Locale.ENGLISH), is(equalTo(HELLO_WORLD_ENGLISH)));
87          collector.checkThat(multi.getValue(Locale.GERMAN), is(equalTo(HELLO_WORLD_GERMAN)));
88          collector.checkThat(multi.getValues().size(), is(equalTo(2)));
89          collector.checkThat(multi.getValues().get(Locale.ENGLISH), is(equalTo(HELLO_WORLD_ENGLISH)));
90          collector.checkThat(multi.getValues().get(Locale.GERMAN), is(equalTo(HELLO_WORLD_GERMAN)));
91          collector.checkThat(multi.hashCode(), is(equalTo(-1027518473)));
92          collector.checkThat(multi_copy.hashCode(), is(equalTo(-1027518473)));
93          collector.checkThat(multi_different.hashCode(), is(equalTo(2008373625)));
94          collector.checkThat(multi, is(equalTo(multi)));
95          collector.checkThat(multi, is(equalTo(multi_copy)));
96          collector.checkThat(multi, is(not(equalTo(null))));
97          collector.checkThat(multi, is(not(equalTo(multi_different))));
98          collector.checkThat(multi.toString(), is(equalTo("Localized[{de=" + HELLO_WORLD_GERMAN + ", " +
99                  "en=" + HELLO_WORLD_ENGLISH + "}]")));
100         collector.checkThat(multi_copy.toString(), is(equalTo("Localized[{de=" + HELLO_WORLD_GERMAN + ", " +
101                 "en=" + HELLO_WORLD_ENGLISH + "}]")));
102         collector.checkThat(multi_different.toString(), is(equalTo("Localized[{de=" + GOODBYE_WORLD_GERMAN + ", " +
103                 "en=" + GOODBYE_WORLD_ENGLISH + "}]")));
104     }
105 }
