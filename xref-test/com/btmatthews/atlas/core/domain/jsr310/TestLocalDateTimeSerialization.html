15  public class TestLocalDateTimeSerialization {
16      private static final String TEST_JSON = "{\"dateTime\":{\"$date\":\"1999-01-01T00:00:00.000Z\"}}";
17  
18      private ObjectMapper objectMapper = new ObjectMapper();
19  
20      @Test
21      public void testDeserializer() throws IOException {
22          final ObjectWithTime obj = objectMapper.readValue(TEST_JSON, ObjectWithTime.class);
23          assertEquals(obj.getDateTime(), LocalDateTime.of(1999, 1, 1, 0, 0, 0, 0));
24      }
25  
26      @Test
27      public void testSerializer() throws JsonProcessingException {
28          final ObjectWithTime obj = new ObjectWithTime(LocalDateTime.of(1999, 1, 1, 0, 0, 0, 0));
29          final String json = objectMapper.writeValueAsString(obj);
30          assertEquals(TEST_JSON, json);
31      }
