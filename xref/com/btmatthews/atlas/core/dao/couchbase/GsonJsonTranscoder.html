3   import com.google.common.base.Charsets;
4   import com.google.gson.Gson;
5   import net.spy.memcached.CachedData;
6   
7   public final class GsonJsonTranscoder<I> extends AbstractJsonTranscoder<I> {
8   
9       public GsonJsonTranscoder(final Class<? extends I> objectClass) {
10          super(objectClass);
11      }
12  
13      @Override
14      public CachedData encode(final I o) {
15          final Gson gson = new Gson();
16          final String json = gson.toJson(o);
17          final byte[] data = json.getBytes(Charsets.UTF_8);
18          return new CachedData(0, data, data.length);
19      }
20  
21      @Override
22      public I decode(final CachedData cachedData) {
23          final Gson gson = new Gson();
24          final String json = new String(cachedData.getData(), Charsets.UTF_8);
25          return gson.fromJson(json, getObjectClass());
26      }
27  }
