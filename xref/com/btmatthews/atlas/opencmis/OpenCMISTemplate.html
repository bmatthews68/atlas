1   package com.btmatthews.atlas.opencmis;
2   
3   import org.apache.chemistry.opencmis.client.api.Session;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;
6   
7   /**
8    * @author <a href="mailto:brian@btmatthews.com">Brian Thomas Matthews</a>
9    * @since 1.0.0
10   */
11  public class OpenCMISTemplate implements OpenCMISAccessor {
12  
13      private static final Logger LOGGER = LoggerFactory.getLogger(OpenCMISTemplate.class);
14  
15      private SessionPool sessionPool;
16  
17      public OpenCMISTemplate() {
18      }
19  
20      public OpenCMISTemplate(final SessionPool pool) {
21          sessionPool = pool;
22      }
23  
24      public void setSessionPool(final SessionPool pool) {
25          sessionPool = pool;
26      }
27  
28      public void execute(String workspaceName, SessionVoidCallback callback) {
29          try {
30              Session session = sessionPool.borrowSession(workspaceName);
31              try {
32                  callback.doInSession(session);
33              } finally {
34                  sessionPool.returnSession(workspaceName, session);
35              }
36          } catch (final Exception e) {
37              final String message = e.getMessage();
38              LOGGER.debug(message, e);
39              throw new RepositoryAccessException(message, e);
40          }
41      }
42  
43      public <T> T execute(String workspaceName, SessionCallback<T> callback) {
44          try {
45              final Session session = sessionPool.borrowSession(workspaceName);
46              try {
47                  return callback.doInSession(session);
48              } finally {
49                  sessionPool.returnSession(workspaceName, session);
50              }
51          } catch (final Exception e) {
52              final String message = e.getMessage();
53              LOGGER.debug(message, e);
54              throw new RepositoryAccessException(message, e);
55          }
56      }
57  
58  }
