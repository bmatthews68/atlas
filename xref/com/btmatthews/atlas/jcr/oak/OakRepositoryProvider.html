3   import com.btmatthews.atlas.jcr.RepositoryProvider;
4   import com.mongodb.DB;
5   import com.mongodb.MongoClient;
6   import org.apache.jackrabbit.oak.jcr.Jcr;
7   import org.apache.jackrabbit.oak.plugins.document.DocumentMK;
8   import org.apache.jackrabbit.oak.plugins.document.DocumentNodeStore;
9   import org.apache.jackrabbit.oak.spi.state.NodeStore;
10  import org.springframework.beans.factory.annotation.Autowired;
11  import org.springframework.beans.factory.annotation.Value;
12  import org.springframework.stereotype.Component;
13  
14  import javax.jcr.Repository;
15  
16  @Component
17  public class OakRepositoryProvider implements RepositoryProvider {
18  
19      private final MongoClient mongoClient;
20  
21      private final String databaseName;
22  
23      @Autowired
24      public OakRepositoryProvider(final MongoClient mongoClient,
25                                   @Value("com.btmatthews.atlas.oak.db.name") final String databaseName) {
26          this.mongoClient = mongoClient;
27          this.databaseName = databaseName;
28      }
29  
30      @Override
31      public Repository getRepository() {
32          final DB db = mongoClient.getDB(databaseName);
33          final NodeStore store = new DocumentNodeStore(new DocumentMK.Builder().setMongoDB(db));
34          return new Jcr(store).createRepository();
35      }
36  }
